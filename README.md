```python
    import random

    arr = random.sample(range(1, 100), 5) 
        print("정렬 전 :", arr) 

        def bubbleSort(arr):
            for i in range(len(arr)-1, 0, -1):
                for j in range(0, i):
                    if arr[j] > arr[j+1]:
                    arr[j], arr[j+1] = arr[j+1], arr[j] # 자리 바꿈
                print(arr)
            return arr
 
        print("정렬 후 :", bubbleSort(arr))
```
